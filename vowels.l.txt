%{
#include <stdio.h>

int vow_count = 0;
int cons_count = 0;
%}

%%
[aeiouAEIOU]                  { vow_count++; }
[b-df-hj-np-tv-zB-DF-HJ-NP-TV-Z] { cons_count++; }
[ \t\n]                       ;   /* ignore spaces and new lines */
.                             ;   /* ignore other characters */
%%

int yywrap()
{
    return 1;
}

int main()
{
    printf("Enter the string:\n");
    yylex();
    printf("Number of vowels = %d\n", vow_count);
    printf("Number of consonants = %d\n", cons_count);
    return 0;
}
